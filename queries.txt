Q11: Check feedback rating and display performance level using IF-ELSIF 

-- Project: University Volunteer Activity Management System 

-- Semester: Spring 2025-2026 

-- Course: Advance Database Management System 

-- Section: (A) 

DECLARE 

    v_rating  NUMBER; 

    v_result  VARCHAR2(20); 

BEGIN 

    SELECT FeedbackRating INTO v_rating 

    FROM Participation WHERE ParticipationID = 1; 

     

    IF v_rating = 5 THEN 

        v_result := 'Excellent'; 

    ELSIF v_rating = 4 THEN 

        v_result := 'Good'; 

    ELSIF v_rating = 3 THEN 

        v_result := 'Average'; 

    ELSE 

        v_result := 'Poor'; 

    END IF; 

     

    DBMS_OUTPUT.PUT_LINE('Rating : ' || v_rating); 

    DBMS_OUTPUT.PUT_LINE('Level  : ' || v_result); 

END;

Q3: Procedure to display all tasks with their status 

 
-- Project: University Volunteer Activity Management System 
-- Semester: Spring 2025-2026 
-- Course: Advance Database Management System 
-- Section: (A) 
 

CREATE OR REPLACE PROCEDURE show_all_tasks IS 
    CURSOR c_task IS SELECT TaskID, TaskName, Status FROM Task; 
    v_id   Task.TaskID%TYPE; 
    v_name Task.TaskName%TYPE; 
    v_stat Task.Status%TYPE; 
BEGIN 
    OPEN c_task; 
    LOOP 
        FETCH c_task INTO v_id, v_name, v_stat; 
        EXIT WHEN c_task%NOTFOUND; 
        DBMS_OUTPUT.PUT_LINE('TaskID: '||v_id||' | '||v_name||' | '||v_stat); 
    END LOOP; 
    CLOSE c_task; 
END; 
/ 

BEGIN 

    show_all_tasks; 

END;  

 

Q4: Procedure to update task status by TaskID 
-- Project: University Volunteer Activity Management System 
-- Semester: Spring 2025-2026 
-- Course: Advance Database Management System 
-- Section: (A) 
 

CREATE OR REPLACE PROCEDURE update_task_status( 
    p_taskid IN NUMBER, p_status IN VARCHAR2) IS 
BEGIN 
    UPDATE Task SET Status = p_status WHERE TaskID = p_taskid; 
    IF SQL%ROWCOUNT = 0 THEN 
        DBMS_OUTPUT.PUT_LINE('No task found with ID: ' || p_taskid); 
    ELSE 
        DBMS_OUTPUT.PUT_LINE('Task '||p_taskid||' updated to: '||p_status); 
        COMMIT; 
    END IF; 
END; 
/ 

BEGIN 
update_task_status(1, 'Completed'); 
END;

BEFORE INSERT OR UPDATE ON Participation

        RAISE_APPLICATION_ERROR(-20001,

-- Package Specification: 
CREATE OR REPLACE PACKAGE pkg_volunteer AS 
    FUNCTION  get_vol_name(p_volid IN NUMBER) RETURN VARCHAR2; 
    PROCEDURE show_vol_skills(p_volid IN NUMBER); 
END pkg_volunteer; 
/ 

-- Package Body: 
CREATE OR REPLACE PACKAGE BODY pkg_volunteer AS 
    FUNCTION get_vol_name(p_volid IN NUMBER) RETURN VARCHAR2 IS 
        v_name VARCHAR2(50); 
    BEGIN 
        SELECT VName INTO v_name FROM Volunteer WHERE VolID=p_volid; 
        RETURN v_name; 
    END; 
    PROCEDURE show_vol_skills(p_volid IN NUMBER) IS 
        CURSOR c_skill IS 
            SELECT Skill FROM Vol_Skill WHERE VolID=p_volid; 
        v_skill Vol_Skill.Skill%TYPE; 
    BEGIN 
        DBMS_OUTPUT.PUT_LINE('Skills of: '||get_vol_name(p_volid)); 
        OPEN c_skill; 
        LOOP 
            FETCH c_skill INTO v_skill; 
            EXIT WHEN c_skill%NOTFOUND; 
            DBMS_OUTPUT.PUT_LINE(' - ' || v_skill); 
        END LOOP; 
        CLOSE c_skill; 
    END; 
END pkg_volunteer; 
/ 

SELECT pkg_volunteer.get_vol_name(1) FROM DUAL; 
BEGIN 
    pkg_volunteer.show_vol_skills(1); 
END;

-- Semester: Spring 2025-2026

-- Package Specification: 
CREATE OR REPLACE PACKAGE pkg_event AS 
    FUNCTION  get_event_name(p_eventid IN NUMBER) RETURN VARCHAR2; 
    PROCEDURE show_event_volunteers(p_eventid IN NUMBER); 
END pkg_event; 
/ 

-- Package Body: 
CREATE OR REPLACE PACKAGE BODY pkg_event AS 
    FUNCTION get_event_name(p_eventid IN NUMBER) RETURN VARCHAR2 IS 
        v_name VARCHAR2(50); 
    BEGIN 
        SELECT EventName INTO v_name FROM Event WHERE EventID=p_eventid; 
        RETURN v_name; 
    END; 
    PROCEDURE show_event_volunteers(p_eventid IN NUMBER) IS 
        CURSOR c_vols IS 
            SELECT V.VName, P.HoursWorked, P.AttStatus 
            FROM Volunteer V, Participation P 
            WHERE V.VolID=P.VolID AND P.EventID=p_eventid; 
        v_rec c_vols%ROWTYPE; 
    BEGIN 
        DBMS_OUTPUT.PUT_LINE('Volunteers: '||get_event_name(p_eventid)); 
        OPEN c_vols; 
        LOOP 
            FETCH c_vols INTO v_rec; 
            EXIT WHEN c_vols%NOTFOUND; 
            DBMS_OUTPUT.PUT_LINE(' - '||v_rec.VName||' | '||v_rec.HoursWorked||'h'); 
        END LOOP; 
        CLOSE c_vols; 
    END; 
END pkg_event; 
/ 

SELECT pkg_event.get_event_name(1) FROM DUAL; 
BEGIN 
pkg_event.show_event_volunteers(1); 
END; 
/
